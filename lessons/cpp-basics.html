3.1 Input/Output Operations

Basic Input/Output in C

C provides standard library functions for input and output operations through the stdio.h header file. The most commonly used functions are printf() for output and scanf() for input.

Example: Basic I/O Operations
/* Basic Input and Output */
#include <stdio.h>

int main() {
    int age;
    float height;
    char name[50];
    
    /* Output using printf */
    printf("Enter your name: ");
    scanf("%s", name);
    
    printf("Enter your age: ");
    scanf("%d", &age);
    
    printf("Enter your height (in meters): ");
    scanf("%f", &height);
    
    /* Display the values */
    printf("\nName: %s\n", name);
    printf("Age: %d years\n", age);
    printf("Height: %.2f meters\n", height);
    
    return 0;
}

Format Specifiers

  • %d - Integer (int)
  • %f - Float
  • %lf - Double
  • %c - Character
  • %s - String
  • %p - Pointer address
3.2 Control Structures

Conditional Statements

If-Else Statement
/* If-Else Example */
#include <stdio.h>

int main() {
    int marks;
    
    printf("Enter marks: ");
    scanf("%d", &marks);
    
    if (marks >= 90) {
        printf("Grade: A+\n");
    }
    else if (marks >= 80) {
        printf("Grade: A\n");
    }
    else if (marks >= 70) {
        printf("Grade: B\n");
    }
    else if (marks >= 60) {
        printf("Grade: C\n");
    }
    else {
        printf("Grade: F\n");
    }
    
    return 0;
}
Switch Statement
/* Switch Case Example */
#include <stdio.h>

int main() {
    int choice;
    
    printf("Menu:\n");
    printf("1. Add\n");
    printf("2. Subtract\n");
    printf("3. Multiply\n");
    printf("Enter choice: ");
    scanf("%d", &choice);
    
    switch(choice) {
        case 1:
            printf("Addition selected\n");
            break;
        case 2:
            printf("Subtraction selected\n");
            break;
        case 3:
            printf("Multiplication selected\n");
            break;
        default:
            printf("Invalid choice\n");
    }
    
    return 0;
}

Loops

For Loop
/* For Loop Example */
#include <stdio.h>

int main() {
    int i;
    int sum = 0;
    
    /* Sum of first 10 numbers */
    for(i = 1; i <= 10; i++) {
        sum = sum + i;
        printf("i = %d, sum = %d\n", i, sum);
    }
    
    printf("\nTotal sum: %d\n", sum);
    
    return 0;
}
While Loop
/* While Loop Example */
#include <stdio.h>

int main() {
    int num = 1;
    
    /* Print numbers 1 to 5 */
    while(num <= 5) {
        printf("Number: %d\n", num);
        num++;
    }
    
    return 0;
}
Do-While Loop
/* Do-While Loop Example */
#include <stdio.h>

int main() {
    int num;
    
    do {
        printf("Enter a positive number (0 to exit): ");
        scanf("%d", &num);
        
        if(num > 0) {
            printf("You entered: %d\n", num);
        }
    } while(num != 0);
    
    printf("Program exited.\n");
    
    return 0;
}
3.3 Functions and Parameters

Function Basics

Functions are reusable blocks of code that perform specific tasks. They help organize code and make it more modular and maintainable.

Function Declaration and Definition
/* Function Examples */
#include <stdio.h>

/* Function declaration (prototype) */
int add(int a, int b);
void greet(char name[]);
int factorial(int n);

int main() {
    int result;
    
    /* Calling functions */
    result = add(10, 20);
    printf("10 + 20 = %d\n", result);
    
    greet("Alice");
    
    result = factorial(5);
    printf("Factorial of 5 = %d\n", result);
    
    return 0;
}

/* Function definition */
int add(int a, int b) {
    return a + b;
}

void greet(char name[]) {
    printf("Hello, %s!\n", name);
}

int factorial(int n) {
    int i, result = 1;
    
    for(i = 1; i <= n; i++) {
        result = result * i;
    }
    
    return result;
}

Pass by Value vs Pass by Reference

Pass by Value
/* Pass by Value - Copy of value is passed */
#include <stdio.h>

void modifyValue(int x) {
    x = x + 10;
    printf("Inside function: %d\n", x);
}

int main() {
    int num = 5;
    
    printf("Before function: %d\n", num);
    modifyValue(num);
    printf("After function: %d\n", num);
    
    /* Output shows num is unchanged */
    return 0;
}
Pass by Reference (using pointers)
/* Pass by Reference - Address is passed */
#include <stdio.h>

void modifyValue(int *x) {
    *x = *x + 10;
    printf("Inside function: %d\n", *x);
}

int main() {
    int num = 5;
    
    printf("Before function: %d\n", num);
    modifyValue(&num);
    printf("After function: %d\n", num);
    
    /* Output shows num is changed */
    return 0;
}
Note: When you pass by value, the function receives a copy of the variable. When you pass by reference (using pointers), the function receives the address of the variable and can modify the original value.

Recursive Functions

Recursion Example
/* Recursive Function */
#include <stdio.h>

int factorial(int n) {
    if(n == 0 || n == 1) {
        return 1;
    }
    else {
        return n * factorial(n - 1);
    }
}

int fibonacci(int n) {
    if(n == 0) {
        return 0;
    }
    else if(n == 1) {
        return 1;
    }
    else {
        return fibonacci(n - 1) + fibonacci(n - 2);
    }
}

int main() {
    int i;
    
    printf("Factorial of 5: %d\n", factorial(5));
    
    printf("Fibonacci sequence (first 10 numbers):\n");
    for(i = 0; i < 10; i++) {
        printf("%d ", fibonacci(i));
    }
    printf("\n");
    
    return 0;
}
3.4 Pointers

Introduction to Pointers

A pointer is a variable that stores the memory address of another variable. Pointers are powerful features in C that allow direct memory manipulation and efficient data structure implementation.

Basic Pointer Operations
/* Pointer Basics */
#include <stdio.h>

int main() {
    int num = 42;
    int *ptr;
    
    /* Assign address of num to pointer */
    ptr = #
    
    printf("Value of num: %d\n", num);
    printf("Address of num: %p\n", &num);
    printf("Value of ptr: %p\n", ptr);
    printf("Value pointed to by ptr: %d\n", *ptr);
    
    /* Modify value using pointer */
    *ptr = 100;
    printf("\nAfter modification:\n");
    printf("Value of num: %d\n", num);
    
    return 0;
}

Pointer Operators

  • & - Address operator (gets memory address)
  • * - Dereference operator (gets value at address)
  • int *ptr - Declares a pointer to an integer

Pointers and Arrays

Array and Pointer Relationship
/* Pointers and Arrays */
#include <stdio.h>

int main() {
    int arr[5] = {10, 20, 30, 40, 50};
    int *ptr;
    int i;
    
    /* Array name is a pointer to first element */
    ptr = arr;
    
    printf("Using array notation:\n");
    for(i = 0; i < 5; i++) {
        printf("arr[%d] = %d\n", i, arr[i]);
    }
    
    printf("\nUsing pointer arithmetic:\n");
    for(i = 0; i < 5; i++) {
        printf("Value: %d, Address: %p\n", *ptr, ptr);
        ptr++;
    }
    
    return 0;
}

Pointers and Functions

Swapping Values Using Pointers
/* Swap function using pointers */
#include <stdio.h>

void swap(int *a, int *b) {
    int temp;
    temp = *a;
    *a = *b;
    *b = temp;
}

int main() {
    int x = 10, y = 20;
    
    printf("Before swap: x = %d, y = %d\n", x, y);
    
    swap(&x, &y);
    
    printf("After swap: x = %d, y = %d\n", x, y);
    
    return 0;
}

Dynamic Memory Allocation

Using malloc and free
/* Dynamic Memory Allocation */
#include <stdio.h>
#include <stdlib.h>

int main() {
    int *arr;
    int n, i;
    
    printf("Enter number of elements: ");
    scanf("%d", &n);
    
    /* Allocate memory dynamically */
    arr = (int*)malloc(n * sizeof(int));
    
    /* Check if memory allocation was successful */
    if(arr == NULL) {
        printf("Memory allocation failed!\n");
        return 1;
    }
    
    /* Input elements */
    printf("Enter %d elements:\n", n);
    for(i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    
    /* Display elements */
    printf("You entered: ");
    for(i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
    
    /* Free allocated memory */
    free(arr);
    
    return 0;
}

Memory Management Functions

  • malloc(size) - Allocates memory of specified size
  • calloc(n, size) - Allocates memory and initializes to zero
  • realloc(ptr, size) - Resizes previously allocated memory
  • free(ptr) - Deallocates previously allocated memory

Structures and Pointers

Pointer to Structure
/* Structure with Pointers */
#include <stdio.h>
#include <string.h>

struct Student {
    int id;
    char name[50];
    float gpa;
};

void displayStudent(struct Student *s) {
    printf("ID: %d\n", s->id);
    printf("Name: %s\n", s->name);
    printf("GPA: %.2f\n", s->gpa);
}

int main() {
    struct Student student1;
    struct Student *ptr;
    
    /* Initialize student */
    student1.id = 101;
    strcpy(student1.name, "John Doe");
    student1.gpa = 3.75;
    
    /* Pointer to structure */
    ptr = &student1;
    
    printf("Student Information:\n");
    displayStudent(ptr);
    
    /* Accessing members using pointer */
    printf("\nModifying using pointer...\n");
    ptr->gpa = 3.90;
    
    printf("\nUpdated Information:\n");
    displayStudent(ptr);
    
    return 0;
}
Note: The arrow operator (->) is used to access structure members through a pointer. It's equivalent to (*ptr).member.

Function Pointers

Using Function Pointers
/* Function Pointers */
#include <stdio.h>

int add(int a, int b) {
    return a + b;
}

int subtract(int a, int b) {
    return a - b;
}

int multiply(int a, int b) {
    return a * b;
}

int main() {
    /* Declare function pointer */
    int (*operation)(int, int);
    int result;
    
    /* Point to add function */
    operation = &add;
    result = operation(10, 5);
    printf("Addition: %d\n", result);
    
    /* Point to subtract function */
    operation = &subtract;
    result = operation(10, 5);
    printf("Subtraction: %d\n", result);
    
    /* Point to multiply function */
    operation = &multiply;
    result = operation(10, 5);
    printf("Multiplication: %d\n", result);
    
    return 0;
}

Common Pointer Mistakes

1. Uninitialized Pointers:

/* WRONG - Dangerous! */
int *ptr;
*ptr = 10;  /* ptr points to random location */

/* CORRECT */
int num;
int *ptr = #
*ptr = 10;

2. Memory Leaks:

/* WRONG - Memory leak */
int *ptr = (int*)malloc(5 * sizeof(int));
/* ... use ptr ... */
/* Forgot to free! */

/* CORRECT */
int *ptr = (int*)malloc(5 * sizeof(int));
/* ... use ptr ... */
free(ptr);

3. Dangling Pointers:

/* WRONG - Dangling pointer */
int *ptr = (int*)malloc(sizeof(int));
free(ptr);
*ptr = 10;  /* Accessing freed memory! */

/* CORRECT */
int *ptr = (int*)malloc(sizeof(int));
free(ptr);
ptr = NULL;  /* Set to NULL after freeing */
Summary

Key Takeaways

  • Input/Output: Use printf() and scanf() with appropriate format specifiers
  • Control Structures: if-else, switch, for, while, and do-while for program flow control
  • Functions: Modular code blocks that can be reused; support recursion
  • Pass by Value: Function receives a copy of the argument
  • Pass by Reference: Function receives the address of the argument (using pointers)
  • Pointers: Variables that store memory addresses; essential for dynamic memory and data structures
  • Dynamic Memory: malloc(), calloc(), realloc(), and free() for runtime memory allocation
  • Memory Management: Always free allocated memory to prevent memory leaks
Important: Understanding these C programming fundamentals is crucial for implementing data structures and algorithms efficiently. Pointers, in particular, are essential for linked lists, trees, and other dynamic data structures.
